📁 Estrutura do Sistema

widget-all/
├── 📁 api/                    # Backend Node.js
│   ├── server.js             # Servidor principal da API
│   ├── package.json          # Dependências
│   └── .env                  # Configurações
├── 📁 modules/               # Módulos compartilhados
│   ├── data-manager.js       # Gerenciamento de dados
│   └── auth-manager.js       # Autenticação MetaMask
├── 📁 contracts/             # Smart Contracts Solidity
│   └── UniversalSaleContract.sol
├── 📁 pages/                 # Frontend
│   ├── index.html            # Landing page
│   └── dashboard.html        # Dashboard do usuário
├── 📁 src/                   # Widget embeddable
│   └── widget-sale.js        # Widget principal
├── 📁 data/                  # Armazenamento (JSON)
│   ├── users.json            # Usuários
│   ├── widgets.json          # Widgets
│   ├── transactions.json     # Transações
│   ├── credits.json          # Créditos
│   └── system_stats.json     # Estatísticas
├── 📁 shared/                # Utilitários
│   ├── metamask-connector.js
│   └── blockchain-utils.js
├── 📄 README.md              # Documentação completa
├── 📄 .env.example           # Exemplo de configuração
└── 📄 start.ps1              # Script de inicialização
